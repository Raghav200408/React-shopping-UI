How to Display in React

jsx
import { Gents } from "./data";

function GentsCard() {
  return (
    <div>
      <h2>{Gents.title}</h2>

      <div className="product-grid">
        {Gents.products.map((item, index) => (
          <div key={index} className="product-box">
            <img src={item.image} alt="mens wear" />
            <p> {item.price}</p>
          </div>
        ))}
      </div>
    </div>
  );
}

export default GentsCard;
